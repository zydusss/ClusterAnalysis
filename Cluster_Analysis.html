Kmeans Clustering

wss=(nrow(mtcars.scale)-1)*sum(apply(mtcars.scale,2,var))

for(i in 2:15)
{
  wss[i]=sum(kmeans(mtcars.scale, centers=i)$withinss)
  }

set.seed(20)
plot(1:15,wss, type="b", xlab="No. of Clusters", ylab="Within Group Sum of Squares ")
